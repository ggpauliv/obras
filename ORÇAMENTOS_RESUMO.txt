🚀 ENDPOINTS DA API:
─────────────────────────────────────────────────────────────────────────────

GET    /api/orcamentos
       Lista orçamentos da obra
       Query: ?obraId=uuid
       Response: { id, nome, fornecedor_nome, valor_total, ... }[]

GET    /api/orcamentos/:id
       Obtém detalhes completos + linhas do orçamento
       Response: { ...orcamento, linhas: [] }

POST   /api/orcamentos
       Criar orçamento manualmente
       Body: { obraId, fornecedorId, nome, valorTotal, prazoDias, ... }
       Response: { id, ... }

DELETE /api/orcamentos/:id
       Remover orçamento
       Response: { sucesso: true }

POST   /api/orcamentos/importar
       ⭐ Importar Excel (base64)
       Body: { obraId, arquivo: "base64..." }
       Response: { sucesso, orcamento, avisos }

GET    /api/orcamentos/comparar
       Análise comparativa
       Query: ?obraId=uuid
       Response: { totalOrcamentos, orcamentos, melhorPreco, diferenca, economia% }

💾 BANCO DE DADOS:
─────────────────────────────────────────────────────────────────────────────

Tabela: orcamentos
  id (UUID)
  obra_id (FK → obras)
  fornecedor_id (FK → fornecedores)
  nome VARCHAR(255)
  descricao TEXT
  valor_total NUMERIC(12,2)
  prazo_dias INTEGER
  status VARCHAR(50)  [ativo|vencido|aceito|descartado]
  data_envio DATE
  data_emissao DATE
  numero_cotacao VARCHAR(100)
  criado_em TIMESTAMP
  atualizado_em TIMESTAMP

Tabela: linhas_orcamento
  id (UUID)
  orcamento_id (FK → orcamentos) — DELETE CASCADE
  item_numero VARCHAR(50)
  descricao TEXT
  quantidade NUMERIC(10,4)
  valor_unitario NUMERIC(12,2)
  valor_total NUMERIC(12,2)
  categoria VARCHAR(100)
  criado_em TIMESTAMP

Índices:
  idx_orcamentos_obra_id
  idx_orcamentos_fornecedor_id
  idx_linhas_orcamento_orcamento_id
  idx_linhas_orcamento_item
